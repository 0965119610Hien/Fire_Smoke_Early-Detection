\section{Tổng quan Hệ thống}

\subsection{Chiến lược Huấn luyện 2 Giai đoạn}

TimeSformer sở hữu hai loại attention hoàn toàn độc lập trong mỗi block:

\begin{table}[h!]
\centering
\begin{tabular}{lll}
\toprule
\textbf{Loại Attention} & \textbf{Học gì} & \textbf{Dữ liệu cần} \\
\midrule
Space Attention & Hình dạng, màu sắc, texture lửa/khói trong 1 frame & Ảnh tĩnh \\
Time Attention  & Biến đổi, chuyển động, lan rộng qua nhiều frame    & Video clip \\
\bottomrule
\end{tabular}
\caption{Hai loại attention trong TimeSformer}
\end{table}

Hệ thống được thiết kế thành \textbf{2 giai đoạn huấn luyện tuần tự}:

\begin{tcolorbox}[
  colback=spatialBlue!5, colframe=spatialBlue!50,
  arc=5pt, boxrule=0.8pt, title={\textbf{Giai đoạn 1 --- Spatial Pre-training}},
  fonttitle=\bfseries\color{spatialBlue}
]
\texttt{num\_frames = 1} $\Rightarrow$ Time Attention gần như vô nghĩa (T=1)\\
\textbf{Dữ liệu:} Ảnh tĩnh YOLO format (\texttt{merged\_dataset/})\\
\textbf{Mục tiêu:} Học nhận biết hình dạng/texture của lửa \& khói\\
\textbf{Output:} \texttt{spatial\_fire\_smoke\_weights\_v2.pth}
\end{tcolorbox}

\vspace{0.3cm}
\begin{center}
  $\downarrow$ \textit{Transfer Learning (strict=False) + Resize time\_embedding T=1$\to$T=8}
\end{center}
\vspace{0.3cm}

\begin{tcolorbox}[
  colback=temporalGreen!5, colframe=temporalGreen!50,
  arc=5pt, boxrule=0.8pt, title={\textbf{Giai đoạn 2 --- Full Spatio-Temporal Training}},
  fonttitle=\bfseries\color{temporalGreen}
]
\texttt{num\_frames = 8} $\Rightarrow$ Time Attention được kích hoạt hoàn toàn\\
\textbf{Dữ liệu:} Video clips theo thư mục \texttt{fire/}, \texttt{smoke/}, \texttt{normal/}\\
\textbf{Mục tiêu:} Học biến đổi không-thời gian của lửa/khói\\
\textbf{Output:} \texttt{full\_fire\_smoke\_weights\_v1.pth}
\end{tcolorbox}

\paragraph{Lý do 2 giai đoạn hiệu quả hơn train end-to-end:}
\begin{itemize}
  \item Dữ liệu ảnh tĩnh dồi dào $\Rightarrow$ spatial features được học tốt từ đầu
  \item Fine-tune temporal chỉ cần dataset video nhỏ hơn nhiều
  \item Tránh gradient vanishing khi train từ đầu với chuỗi frames dài
\end{itemize}

\subsection{Stochastic Depth (DropPath)}

Tỷ lệ drop tăng tuyến tính theo chiều sâu của mạng, từ $0$ đến $0.1$:

\[
\text{dpr}_i = \frac{i}{L-1} \times r_{\max}, \quad i=0,1,\ldots,L-1, \quad L=6, \quad r_{\max}=0.1
\]

\begin{center}
\begin{tikzpicture}
  \foreach \i in {0,...,5} {
    \pgfmathsetmacro{\dpr}{\i * 0.02}

\subsection{Tại sao temporal\_fc init zeros là quan trọng?}

\begin{tcolorbox}[
  colback=firebrick!5, colframe=firebrick!40,
  arc=4pt, boxrule=0.6pt
]
\[
\text{Temporal residual} = \texttt{temporal\_fc}\!\left(\texttt{temporal\_attn}(x)\right)
= \mathbf{W}_{\text{temp}} \cdot (\ldots) + \mathbf{b} = \mathbf{0}
\quad (\text{vì } \mathbf{W}=0, \, \mathbf{b}=0)
\]

\textbf{Hệ quả:} Ở epoch đầu tiên, temporal contribution $= 0$ $\Rightarrow$ model hoàn toàn giống Giai đoạn 1 (spatial only) $\Rightarrow$ Không phá vỡ spatial features đã học. Temporal\_fc sẽ dần học ra giá trị khác 0 theo gradient descent.
\end{tcolorbox}
